Private Sub 印刷実行(num As Integer)
'num 番目にデータをコピーして印刷を行う
Dim i As Integer
Dim db As Database
Dim rs As Recordset

    Set db = CurrentDb
    Set rs = db.OpenRecordset("1枚印刷用テーブル", dbOpenDynaset)
    'データの初期化
    rs.MoveFirst
    Do Until rs.EOF
        rs.Edit
        rs![発送先氏名] = "  "
        rs![発送先敬称] = "  "
        rs![発送先郵便番号] = "  "
        rs![発送先住所1] = "  "
        rs![発送先住所2] = "  "
        rs![発送先住所3] = ""
        rs.Update
        rs.MoveNext
    Loop
    '必要な場所にデータを書込む
    rs.FindFirst "ID=" & num
    If rs.NoMatch Then
        MsgBox "内部のプログラムの異常です"
    End If
    rs.Edit
    rs![発送先氏名] = Me![氏名]
    rs![発送先敬称] = Me![敬称]
    rs![発送先郵便番号] = "〒" & Left(Me![郵便番号], 3) & "-" & Mid(Me![郵便番号], 4, 4)
    rs![発送先住所1] = Me![住所1]
    rs![発送先住所2] = Me![住所2]
    rs![発送先住所3] = Me![住所3]
    
    rs.Update
    rs.Close
    db.Close
    DoCmd.OpenReport "ラベル1枚印刷", acPreview
End Sub
